import pandas as pd
import numpy as np

import seaborn as sns

# Load dataset
df = pd.read_csv("ecommerce.csv")

# Clean data

df.dropna(subset=['user_id', 'price'], inplace=True)
df['price'] = pd.to_numeric(df['price'], errors='coerce')
df['purchase_date'] = pd.to_datetime(df['purchase_date'])

ü§ñ Bonus: ML Task

from sklearn.model_selection import train_test_split

# Predict price based on age and category

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
model = LinearRegression()
model.fit(X_train, y_train)

# Step-by-step from load to ML
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

df = pd.read_csv("ecommerce.csv")
df.drop_duplicates(inplace=True)
df.dropna(subset=['user_id', 'price'], inplace=True)
df['price'] = pd.to_numeric(df['price'], errors='coerce')
df['purchase_date'] = pd.to_datetime(df['purchase_date'])

sns.histplot(df['price'], kde=True)
plt.title("Price Distribution")
plt.show()

sns.barplot(x='category', y='price', data=df, estimator=np.sum)
plt.title("Total Sales by Category")
plt.xticks(rotation=45)
plt.show()

df['month'] = df['purchase_date'].dt.to_period('M')
monthly_sales = df.groupby('month')['price'].sum()

pivot = pd.pivot_table(df, values='price', index='gender', columns='category', aggfunc='mean')

df['category_encoded'] = df['category'].astype('category').cat.codes
X = df[['age', 'category_encoded']]
y = df['price']
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
model = LinearRegression()
model.fit(X_train, y_train)
predictions = model.predict(X_test)
print("MSE:", mean_squared_error(y_test, predictions))
