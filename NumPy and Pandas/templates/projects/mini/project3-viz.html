import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Load and clean
df = pd.read_csv("students.csv")
df = df.dropna()
df['score'] = pd.to_numeric(df['score'], errors='coerce')
df['hours_studied'] = pd.to_numeric(df['hours_studied'], errors='coerce')

# Histogram
sns.histplot(df['score'], kde=True, bins=15, color='skyblue')
plt.title("Score Distribution")
plt.xlabel("Score")
plt.ylabel("Frequency")
plt.grid(True)
plt.show()

# Boxplot
sns.boxplot(x='gender', y='score', data=df, palette='Set2')
plt.title("Score Distribution by Gender")
plt.show()

# Scatter Plot
sns.scatterplot(x='hours_studied', y='score', hue='gender', data=df)
plt.title("Hours Studied vs Score")
plt.xlabel("Hours Studied")
plt.ylabel("Score")
plt.grid(True)
plt.show()

# Correlation Heatmap

plt.title("Correlation Heatmap")
plt.show()
